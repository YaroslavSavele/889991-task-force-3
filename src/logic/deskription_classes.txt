class CheckingIsCity {
  Проверка, существует ли город в таблице городов
}

class Role {
  Определяет какая роль у данного пользователя
}

class User {
  Сохраняет данные пользователя полученные из формы регистарации
}

class ChekingIsCategori {
  Проверяет, есть ли такая категория в таблице категорий
}

class Files {
  Переносит сохраненные файлы из формы добавления задания в публичную дирректорию
  и сохраняет ссылки на них
}

class Exsecutor {
  Сохраняет или меняет данные Исполнителя, записывает в таблицу, выводит данные о исполнителе
  на страницу профиля пользователя
}

class Response {
  Сохраняет отклик в таблицу откликов и выводит отклик на страницу показа задания
}

class Review {
  Записывает отзыв в таблицу отзывов, выводит отзыв на страницу Исполнителя, меняет данные Исполнителя в таблице исполнителей - количество выполненных заданий,
  среднюю оценку. Считает текущую статистику пользователя и выводит его рейтинг и статистику на страницу профиль Исполнителя
}
